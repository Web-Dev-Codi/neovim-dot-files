return {
  "matze/vim-move",
  opts = {
    --- Config
  },
  config = function()
    require("move").setup {}
    vim.keymap.set("v", "<C-k>", ":Move current line/selection up<CR>", {})
    vim.keymap.set("v", "<C-j>", ":Move current line/selection down", {})
    vim.keymap.set("v", "<C-h>", ":Move current character/selection left<CR>", {})
    vim.keymap.set("v", "<C-l>", ":Move current character/selection right<CR>", {})
  end,
}
